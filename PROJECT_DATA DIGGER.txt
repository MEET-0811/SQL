1 Customers Table Fields:-
- CustomerID 
- Name 
- Email
- Address

CREATE TABLE Customers(
    Customer_Id INT PRIMARY KEY,
    Name VARCHAR(50),
    Email VARCHAR(50),

    );

2 Orders Table :-

-OrderID

3 Products Table Fields:-
-ProductID
- ProductName
- Price
- Stock 

CREATE TABLE product(
    ProductID INT PRIMARY KEY,
    ProductName VARCHAR(50),
    Price DECIMAL(10,2),
    Stock INT 
    );

* Queries to Perform:-
